    First Terraform State

  • Lets check the Stopped Instances with Terraform and then startup.
  • Check terraform providers.tf file.
    • 
berk-MBP start_stopped % cat provders.tf
terraform {
  required_providers {
	aws = {
	  source = "hashicorp/aws"
	}
  }
}

provider "aws" {
  region  = "eu-west-2"
  profile = "tf-access-user"
}

  • Check terraform main.tf file.
    • 
berk-MBP start_stopped % cat main.tf

data "aws_instances" "stopped_instances" {
  instance_state_names = ["stopped"]
}

resource "null_resource" "start_instances" {
  count = length(data.aws_instances.stopped_instances.ids)
  provisioner "local-exec" {
    command = "aws ec2 start-instances --instance-ids ${data.aws_instances.stopped_instances.ids[count.index]}"
  }
}
